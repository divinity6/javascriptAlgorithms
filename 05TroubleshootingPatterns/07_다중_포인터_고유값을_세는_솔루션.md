
````javascript
  
    function countUniqueValues( arr ){
        
        if ( 0 === arr.length ){
            return 0;
        }    
    
        var i = 0;

        /**
         * - j 는 루프에 의해 자동증가하므로,
         *   i 만 따로 증가시켜주면 된다
         */
        for ( var j = 1; j < arr.length; j++ ){
            if ( arr[ i ] !== arr[ j ] ){
                i++;
                arr[ i ] = arr[ j ];
            }    
        }
        
        return i + 1;
    }

````
